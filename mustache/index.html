Getting started

To connect to the core api is easy, here are some connectors:

CURL commandline

  $ curl -H 'x-{{{name}}}-token: 3909334aba998a90c78a0' -H 'Content-Type: application/json' -X POST {{{host}}}/user --data '{}'

JAVASCRIPT

  var req    = require('request');
  var apikey = "3909334aba998a90c78a0";
  var host   = '{{{host}}}';

  req.get({
    headers: {
      'x-{{{name}}}-token': apikey
    },
    url: host + '/user'
  }, function(error, response, body) {
    // do something with response
    return !error && response.statusCode === 200 && response.body;
  });

COFFEESCRIPT

  req   = require('request')
  apikey = "3909334aba998a90c78a0"
  host  = '{{{host}}}'

  req.get {
    headers: { 'x-{{{name}}}-token': apikey }
    url:host+'/user'
  }, (error, response, body) ->
    ok = (!error and response.statusCode == 200 and response.body)

PHP

  $json = json_encode( (object)array("foo" => "bar") );
  $resource = "/user";
  $ch = curl_init('{{{host}}}'.$resource);
  curl_setopt($ch, CURLOPT_CUSTOMREQUEST, "POST");
  curl_setopt($ch, CURLOPT_POSTFIELDS, $data_string);
  curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
  curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
  curl_setopt($ch, CURLOPT_HTTPHEADER, array(
    'Content-Type: application/json',
    'Content-Length: ' . strlen($data_string))
  );
  $result = curl_exec($ch);
  // HINT: use a REST client like https://github.com/bproctor/rest
  //       or install one using composer: http://getcomposer.org
